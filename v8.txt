class LinkValidationSystem {
  static validateURL(url) {
    // Шаг 1: Базовая структура
    const urlStructure = this.validateURLStructure(url);
    if (!urlStructure.isValid) return this.generateValidationError("Некорректная
структура ссылки");

    // Шаг 2: Доступность
    const availability = this.checkURLAvailability(url);
    if (!availability.isAvailable) return
this.handleAvailabilityError(availability.statusCode);

    // Шаг 3: Безопасность
    const safetyScan = this.scanContentSafety(url);
    if (!safetyScan.isSafe) return
this.handleSafetyViolation(safetyScan.threats);

    // Шаг 4: Качество контента
    const contentQuality = this.assessContentQuality(url);
    if (!contentQuality.isRelevant) return
this.handleLowQualityContent(contentQuality.score);

    // Шаг 5: Расчет доверия
    const trustScore = this.calculateTrustScore(url, availability, safetyScan,
contentQuality);

      return {
        isValid: true,
        url: url,
        trustScore: trustScore,
        category: this.determineContentCategory(url),
        recommendations: this.generateUsageRecommendations(trustScore)
      };
 }

  static generateTrustedLinksSection(links) {
    // Фильтрация и сортировка ссылок по уровню доверия
    const validatedLinks = links.map(link => this.validateURL(link.url))
      .filter(validation => validation.isValid && validation.trustScore.score >=
75)
      .sort((a, b) => b.trustScore.score - a.trustScore.score);

      return {
        trustedLinks: validatedLinks.slice(0, 5), // Максимум 5 надежных ссылок
        rejectedCount: links.length - validatedLinks.length,
        message: validatedLinks.length > 0
          ? `✅ Наидены ${validatedLinks.length} проверенных источников`
          : "❌ Не наидено надежных источников по этои теме"
      };
  }
}
